Пример использования

<script src="/assets/pocketbase.umd.js"></script>
<script type="module">
  import { list, create, update, remove } from "/assets/pocketbase.core.js";

  const page = await list("example", { page: 1, perPage: 10 });
  const created = await create("example", { name: "Ivan", age: 21, isMaried: false });
  const updated = await update("example", created.id, { age: 22 });
  await remove("example", created.id);
</script>
